private void calculateTotal() {
    // 1. Éjszakák számának kiszámítása
    long diffInMs = departureDate.getTimeInMillis() - arrivalDate.getTimeInMillis();
    long nights = TimeUnit.MILLISECONDS.toDays(diffInMs);
    
    // Biztonsági ellenőrzés: ha ugyanaz a nap, legyen 0, de a matek miatt 
    // a feladat valószínűleg legalább 1 éjszakával számol a példában (5 éj).
    if (nights < 0) nights = 0; 

    // 2. Alapadatok lekérése a UI-ról
    int extraReggeli = swBreakfast.isChecked() ? 2000 : 0;
    int szemelyekSzama = sbPeopleCount.getProgress();

    // 3. A PONTOS KÉPLET a feladatlap alapján:
    // Vegösszeg = (Szobaár + (ReggeliÁr * Személyek)) * Éjszakák
    long vegosszeg = (basePrice + (extraReggeli * szemelyekSzama)) * nights;

    // 4. Megjelenítés formázva (pl. 315.000 Ft)
    if (nights > 0) {
        tvTotalPrice.setText("Végösszeg:\n" + String.format("%,d", vegosszeg) + " Ft");
    } else {
        tvTotalPrice.setText("Végösszeg:\n0 Ft (Válasszon érvényes dátumot)");
    }
}

package com.example.hotelapp; // Ügyelj a saját package nevedre!

import android.app.DatePickerDialog;
import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;
import java.util.Calendar;
import java.util.concurrent.TimeUnit;

public class SecondActivity extends AppCompatActivity {

    private TextView tvRoomDetails, tvPeopleLabel, tvTotalPrice;
    private EditText etGuestName;
    private SeekBar sbPeopleCount;
    private Switch swBreakfast;
    private Calendar arrivalDate, departureDate;
    private int basePrice, maxCap;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        // View-k inicializálása
        tvRoomDetails = findViewById(R.id.tvRoomDetails);
        tvPeopleLabel = findViewById(R.id.tvPeopleLabel);
        tvTotalPrice = findViewById(R.id.tvTotalPrice);
        etGuestName = findViewById(R.id.etGuestName);
        sbPeopleCount = findViewById(R.id.sbPeopleCount);
        swBreakfast = findViewById(R.id.swBreakfast);

        // Adatok átvétele az első Activity-től
        String roomName = getIntent().getStringExtra("name");
        basePrice = getIntent().getIntExtra("price", 0);
        maxCap = getIntent().getIntExtra("max", 0);

        tvRoomDetails.setText(roomName + " - " + basePrice + " Ft/éj, max. " + maxCap + " fő");

        // Dátumok alaphelyzetbe állítása (ma és holnap)
        arrivalDate = Calendar.getInstance();
        departureDate = Calendar.getInstance();
        departureDate.add(Calendar.DATE, 1);

        // SeekBar eseménykezelő
        sbPeopleCount.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
            @Override
            public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
                tvPeopleLabel.setText("Személyek száma: " + progress + " fő");
                calculateTotal();
            }
            @Override public void onStartTrackingTouch(SeekBar seekBar) {}
            @Override public void onStopTrackingTouch(SeekBar seekBar) {}
        });

        // Switch eseménykezelő
        swBreakfast.setOnCheckedChangeListener((buttonView, isChecked) -> calculateTotal());

        // Dátum gombok
        findViewById(R.id.btnArrival).setOnClickListener(v -> showDatePicker(true));
        findViewById(R.id.btnDeparture).setOnClickListener(v -> showDatePicker(false));

        // Foglalás gomb + Validáció
        findViewById(R.id.btnSubmit).setOnClickListener(v -> {
            String name = etGuestName.getText().toString().trim();
            int currentPeople = sbPeopleCount.getProgress();

            if (name.isEmpty()) {
                Toast.makeText(this, "Kérem adja meg a nevét!", Toast.LENGTH_SHORT).show();
            } else if (currentPeople > maxCap) {
                Toast.makeText(this, "A szoba max. " + maxCap + " fős! Állítsa be újra.", Toast.LENGTH_LONG).show();
            } else {
                Toast.makeText(this, "Sikeres foglalás " + name + " részére!", Toast.LENGTH_SHORT).show();
                finish(); // Vissza az első oldalra
            }
        });

        calculateTotal(); // Kezdeti számítás
    }

    private void showDatePicker(boolean isArrival) {
        Calendar target = isArrival ? arrivalDate : departureDate;
        DatePickerDialog dpd = new DatePickerDialog(this, (view, year, month, dayOfMonth) -> {
            target.set(year, month, dayOfMonth);
            
            // Logika: Távozás nem lehet érkezés előtt
            if (!isArrival && departureDate.before(arrivalDate)) {
                departureDate.setTimeInMillis(arrivalDate.getTimeInMillis());
                departureDate.add(Calendar.DATE, 1);
                Toast.makeText(this, "A távozás nem lehet korábbi az érkezésnél!", Toast.LENGTH_SHORT).show();
            }
            calculateTotal();
        }, target.get(Calendar.YEAR), target.get(Calendar.MONTH), target.get(Calendar.DAY_OF_MONTH));
        
        dpd.getDatePicker().setMinDate(System.currentTimeMillis());
        dpd.show();
    }

    private void calculateTotal() {
        long diffInMs = departureDate.getTimeInMillis() - arrivalDate.getTimeInMillis();
        long nights = TimeUnit.MILLISECONDS.toDays(diffInMs);
        if (nights <= 0) nights = 1; // Minimum 1 éjszaka

        int breakfastPrice = swBreakfast.isChecked() ? 2000 : 0;
        int people = sbPeopleCount.getProgress();

        // Képlet: (Szobaár + (reggeli * fő)) * éjszakák
        long total = (long) (basePrice + (breakfastPrice * people)) * nights;
        
        tvTotalPrice.setText("Végösszeg:\n" + String.format("%,d", total) + " Ft");
    }
}

import android.app.AlertDialog;

import androidx.appcompat.app.AppCompatActivity;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Random;

public class MainActivity extends AppCompatActivity {

    private TextView tvPontErtek;
    private ImageView ivSzin;
    private ListView lvSzinek;
    private ArrayList<String> szinNevek;
    private ArrayList<Integer> szinKodok;
    private ArrayAdapter<String> adapter;

    private int pontszam = 0;
    private int aktualisSzinIndex = -1; // -1, hogy kezdés előtt ne legyen egyezés
    private Handler handler = new Handler();
    private Runnable runnable;

    // A "KAPUŐR": Ez akadályozza meg, hogy egy körben többször kattintsunk
    private boolean voltMarKattintas = true;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        tvPontErtek = findViewById(R.id.tvPontErtek);
        ivSzin = findViewById(R.id.ivSzin);
        lvSzinek = findViewById(R.id.lvSzinek);

        // Alapszínek inicializálása
        szinNevek = new ArrayList<>(Arrays.asList("fekete", "fehér", "piros", "kék"));
        szinKodok = new ArrayList<>(Arrays.asList(Color.BLACK, Color.WHITE, Color.RED, Color.BLUE));

        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, szinNevek);
        lvSzinek.setAdapter(adapter);

        // Pontozás kezelése
        lvSzinek.setOnItemClickListener((parent, view, position, id) -> {
            if (!voltMarKattintas) { // Csak ha még nem kattintottunk ebben a körben
                if (position == aktualisSzinIndex) {
                    pontszam += 5;
                    Toast.makeText(this, "Helyes! (+5)", Toast.LENGTH_SHORT).show();
                } else {
                    pontszam -= 1;
                    Toast.makeText(this, "Rossz! (-1)", Toast.LENGTH_SHORT).show();
                }
                tvPontErtek.setText(String.valueOf(pontszam));
                voltMarKattintas = true; // Lezárjuk a kört a következő színig
            } else {
                Toast.makeText(this, "Várj a következő színre!", Toast.LENGTH_SHORT).show();
            }
        });

        // Vissza gomb kezelés
        getOnBackPressedDispatcher().addCallback(this, new OnBackPressedCallback(true) {
            @Override
            public void handleOnBackPressed() {
                PopUp();
            }
        });
    }

    public void onStartClick(View view) {
        if (runnable != null) handler.removeCallbacks(runnable);
        runnable = new Runnable() {
            @Override
            public void run() {
                aktualisSzinIndex = new Random().nextInt(szinKodok.size());
                ivSzin.setBackgroundColor(szinKodok.get(aktualisSzinIndex));

                voltMarKattintas = false; // ÚJ SZÍN: Újra lehet kattintani egyszer!

                handler.postDelayed(this, 5000);
            }
        };
        handler.post(runnable);
    }

    public void onSettingsClick(View view) {
        Intent intent = new Intent(this, Activity2.class);
        intent.putExtra("pontszam", pontszam);
        startActivity(intent);
    }

    public void PopUp() {
        LinearLayout layout = new LinearLayout(this);
        layout.setOrientation(LinearLayout.VERTICAL);
        layout.setPadding(40, 40, 40, 40);

        final EditText etNev = new EditText(this);
        etNev.setHint("Szín neve");
        final EditText etHex = new EditText(this);
        etHex.setHint("Hex kód (pl. #00FF00)");

        Button btnOk = new Button(this);
        btnOk.setText("Hozzáad");

        layout.addView(etNev);
        layout.addView(etHex);
        layout.addView(btnOk);

        AlertDialog dialog = new AlertDialog.Builder(this)
                .setTitle("Új szín hozzáadása")
                .setView(layout)
                .setNegativeButton("Mégse", (d, w) -> d.dismiss())
                .create();

        btnOk.setOnClickListener(v -> {
            try {
                String nev = etNev.getText().toString();
                int kod = Color.parseColor(etHex.getText().toString());
                if (!nev.isEmpty()) {
                    szinNevek.add(nev);
                    szinKodok.add(kod);
                    adapter.notifyDataSetChanged();
                    dialog.dismiss();
                }
            } catch (Exception e) {
                Toast.makeText(this, "Hibás hex kód! (Pl: #FF0000)", Toast.LENGTH_SHORT).show();
            }
        });
        dialog.show();
    }
}

// SecondActivity.java
public class SecondActivity extends AppCompatActivity {
    Calendar arrDate = Calendar.getInstance(), depDate = Calendar.getInstance();
    int roomPrice, maxCapacity;
    TextView tvTotal, tvInfo, tvPCount;
    SeekBar sb; Switch sw; EditText et;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        roomPrice = getIntent().getIntExtra("price", 0);
        maxCapacity = getIntent().getIntExtra("max", 0);
        
        tvInfo = findViewById(R.id.tvRoomInfo);
        tvTotal = findViewById(R.id.tvTotal);
        tvPCount = findViewById(R.id.tvPeopleCount);
        sb = findViewById(R.id.sbPeople);
        sw = findViewById(R.id.swBreakfast);
        et = findViewById(R.id.etName);

        tvInfo.setText(getIntent().getStringExtra("name") + " - " + roomPrice + " Ft/éj");

        sb.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
            @Override public void onProgressChanged(SeekBar s, int p, boolean b) { 
                tvPCount.setText("Személyek száma: " + p + " fő"); updatePrice(); 
            }
            @Override public void onStartTrackingTouch(SeekBar s) {} @Override public void onStopTrackingTouch(SeekBar s) {}
        });

        sw.setOnCheckedChangeListener((b, isChecked) -> updatePrice());

        findViewById(R.id.btnArrival).setOnClickListener(v -> showDatePicker(true));
        findViewById(R.id.btnDeparture).setOnClickListener(v -> showDatePicker(false));

        findViewById(R.id.btnBook).setOnClickListener(v -> {
            if(et.getText().toString().isEmpty()) Toast.makeText(this, "Név kötelező!", Toast.LENGTH_SHORT).show();
            else if(sb.getProgress() > maxCapacity) Toast.makeText(this, "Túl sok vendég!", Toast.LENGTH_LONG).show();
            else { Toast.makeText(this, "Sikeres foglalás!", Toast.LENGTH_SHORT).show(); finish(); }
        });
    }

    private void showDatePicker(boolean arrival) {
        DatePickerDialog dpd = new DatePickerDialog(this, (view, y, m, d) -> {
            if(arrival) arrDate.set(y, m, d); else depDate.set(y, m, d);
            updatePrice();
        }, 2026, 3, 2); // Példa dátum a kép alapján
        dpd.getDatePicker().setMinDate(System.currentTimeMillis());
        dpd.show();
    }

    private void updatePrice() {
        long diff = depDate.getTimeInMillis() - arrDate.getTimeInMillis();
        long nights = Math.max(0, diff / (24 * 60 * 60 * 1000));
        int breakfast = sw.isChecked() ? 2000 : 0;
        long total = (roomPrice + (breakfast * sb.getProgress())) * nights;
        tvTotal.setText("Végösszeg: " + total + " Ft");
    }
}
