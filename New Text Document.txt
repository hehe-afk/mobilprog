MAINACTIVITY.JAVA
import android.app.AlertDialog;
import android.content.Intent;
import android.graphics.Color;
import android.os.Bundle;
import android.os.Handler;
import android.view.View;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ImageView;
import android.widget.LinearLayout;
import android.widget.ListView;
import android.widget.TextView;
import android.widget.Toast;
import androidx.activity.OnBackPressedCallback;
import androidx.appcompat.app.AppCompatActivity;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Random;

public class MainActivity extends AppCompatActivity {

    private TextView tvPontErtek;
    private ImageView ivSzin;
    private ListView lvSzinek;
    private ArrayList<String> szinNevek;
    private ArrayList<Integer> szinKodok;
    private ArrayAdapter<String> adapter;

    private int pontszam = 0;
    private int aktualisSzinIndex = -1; // -1, hogy kezdés előtt ne legyen egyezés
    private Handler handler = new Handler();
    private Runnable runnable;

    // A "KAPUŐR": Ez akadályozza meg, hogy egy körben többször kattintsunk
    private boolean voltMarKattintas = true;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        tvPontErtek = findViewById(R.id.tvPontErtek);
        ivSzin = findViewById(R.id.ivSzin);
        lvSzinek = findViewById(R.id.lvSzinek);

        // Alapszínek inicializálása
        szinNevek = new ArrayList<>(Arrays.asList("fekete", "fehér", "piros", "kék"));
        szinKodok = new ArrayList<>(Arrays.asList(Color.BLACK, Color.WHITE, Color.RED, Color.BLUE));

        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, szinNevek);
        lvSzinek.setAdapter(adapter);

        // Pontozás kezelése
        lvSzinek.setOnItemClickListener((parent, view, position, id) -> {
            if (!voltMarKattintas) { // Csak ha még nem kattintottunk ebben a körben
                if (position == aktualisSzinIndex) {
                    pontszam += 5;
                    Toast.makeText(this, "Helyes! (+5)", Toast.LENGTH_SHORT).show();
                } else {
                    pontszam -= 1;
                    Toast.makeText(this, "Rossz! (-1)", Toast.LENGTH_SHORT).show();
                }
                tvPontErtek.setText(String.valueOf(pontszam));
                voltMarKattintas = true; // Lezárjuk a kört a következő színig
            } else {
                Toast.makeText(this, "Várj a következő színre!", Toast.LENGTH_SHORT).show();
            }
        });

        // Vissza gomb kezelés
        getOnBackPressedDispatcher().addCallback(this, new OnBackPressedCallback(true) {
            @Override
            public void handleOnBackPressed() {
                PopUp();
            }
        });
    }

    public void onStartClick(View view) {
        if (runnable != null) handler.removeCallbacks(runnable);
        runnable = new Runnable() {
            @Override
            public void run() {
                aktualisSzinIndex = new Random().nextInt(szinKodok.size());
                ivSzin.setBackgroundColor(szinKodok.get(aktualisSzinIndex));

                voltMarKattintas = false; // ÚJ SZÍN: Újra lehet kattintani egyszer!

                handler.postDelayed(this, 5000);
            }
        };
        handler.post(runnable);
    }

    public void onSettingsClick(View view) {
        Intent intent = new Intent(this, Activity2.class);
        intent.putExtra("pontszam", pontszam);
        startActivity(intent);
    }

    public void PopUp() {
        LinearLayout layout = new LinearLayout(this);
        layout.setOrientation(LinearLayout.VERTICAL);
        layout.setPadding(40, 40, 40, 40);

        final EditText etNev = new EditText(this);
        etNev.setHint("Szín neve");
        final EditText etHex = new EditText(this);
        etHex.setHint("Hex kód (pl. #00FF00)");

        Button btnOk = new Button(this);
        btnOk.setText("Hozzáad");

        layout.addView(etNev);
        layout.addView(etHex);
        layout.addView(btnOk);

        AlertDialog dialog = new AlertDialog.Builder(this)
                .setTitle("Új szín hozzáadása")
                .setView(layout)
                .setNegativeButton("Mégse", (d, w) -> d.dismiss())
                .create();

        btnOk.setOnClickListener(v -> {
            try {
                String nev = etNev.getText().toString();
                int kod = Color.parseColor(etHex.getText().toString());
                if (!nev.isEmpty()) {
                    szinNevek.add(nev);
                    szinKodok.add(kod);
                    adapter.notifyDataSetChanged();
                    dialog.dismiss();
                }
            } catch (Exception e) {
                Toast.makeText(this, "Hibás hex kód! (Pl: #FF0000)", Toast.LENGTH_SHORT).show();
            }
        });
        dialog.show();
    }
}

import android.os.Bundle;
import android.widget.SeekBar;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class Activity2 extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_2);

        TextView tvPont = findViewById(R.id.tvAtvittPont);
        SeekBar sb = findViewById(R.id.sbBetumeret);

        // Pontszám fogadása
        int pontszam = getIntent().getIntExtra("pontszam", 0);
        tvPont.setText("Eddigi pontszámod: " + pontszam);

        // SeekBar kezelése (5-30 tartomány)
        sb.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
            @Override
            public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
                // Mivel a min értéket az XML-ben 5-re állítottuk:
                tvPont.setTextSize(progress);
            }
            @Override public void onStartTrackingTouch(SeekBar seekBar) {}
            @Override public void onStopTrackingTouch(SeekBar seekBar) {}
        });
    }
}

A Start gombbal indítható időzítő:

Java
public void startTimer() {
    timerHandler.postDelayed(new Runnable() {
        @Override
        public void run() {
            if (hatralevoIdo > 0) {
                hatralevoIdo--;
                // tvIdo.setText("Idő: " + hatralevoIdo);
                timerHandler.postDelayed(this, 1000); // 1 másodpercenként fut le
            } else {
                // LEJÁRT AZ IDŐ
                handler.removeCallbacks(runnable); // A színváltós időzítőt is állítsuk le
                Toast.makeText(MainActivity.this, "VÉGE A JÁTÉKNAK!", Toast.LENGTH_LONG).show();
                // Itt átküldheted a júzert az összesítő képernyőre (Activity2)
            }
        }
    }, 1000);
}
